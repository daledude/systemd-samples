# Automatically start a user's systemd session

# Files:
#	user@1000.service
#
# This file is a copy of /usr/lib/systemd/system/user@.service
# with flatcap/1000 substituted in for %I/%i

cp user@1000.service /usr/lib/systemd/system
systemctl enable user@1000.service
systemctl start user@1000.service

# Process started (as flatcap):
#	/usr/lib/systemd/systemd --user
